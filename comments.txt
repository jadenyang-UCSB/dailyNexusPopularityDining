def filterApplication():
    kernel = np.ones((3,3), dtype= np.float32)/9.0
    img = Image.open("blank.jpeg").convert("RGB")

    imageArray = np.array(img)
    height = imageArray.shape[0]
    width = imageArray.shape[1]

    rgbCode = np.zeros((height - 50,width - 50,3),dtype= np.float32)


    for rgb in range(3):
        for i in range(height-50):
            for j in range(width-50):
                rgbCode[i,j,rgb] = np.sum(kernel * imageArray[i:i+3,j:j+3,rgb])
    
    return rgbCode

def compareVector(vec_A, vec_B):
    if(vec_B is None):
        return False
    
    number = np.dot(vec_A,vec_B)
    magA = np.linalg.norm(vec_A)
    magB = np.linalg.norm(vec_B)
    denom = magA * magB
    if denom == 0:
        print("This is always printing")
        return False
    overall = number/denom
    # print(overall)
    
    if(overall >= 0.66):
        return True
    return False


def comparePosition(pos_A, pos_B):
    if(pos_B is None):
        return False
    distance = ((pos_A[0] - pos_B[0])**2 + (pos_A[1] - pos_B[1])**2)**0.5
    
    if(distance < 200):
        return True, distance
    return False, distance
